using System;
using System.Linq;
using System.Windows.Forms;

namespace ArrayOperationsApp
{
    public partial class Form1 : Form
    {
        // Для .NET 6+ делаем поля nullable (могут быть null)
        private int[]? originalArray = null;
        private int[]? sortedArray = null;

        public Form1()
        {
            InitializeComponent();

            // Начальные значения по умолчанию
            nudSize.Value = 10;
            nudMin.Value = -50;
            nudMax.Value = 50;

            // Подключаем обработчики событий
            btnGenerate.Click += btnGenerate_Click;
            btnSort.Click += btnSort_Click;
            btnCalculate.Click += btnCalculate_Click;
        }

        // Генерация массива
        private void btnGenerate_Click(object sender, EventArgs e)
        {
            int size = (int)nudSize.Value;
            int min = (int)nudMin.Value;
            int max = (int)nudMax.Value;

            if (min > max)
            {
                MessageBox.Show("Минимальное значение не может быть больше максимального.",
                                "Ошибка ввода", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }

            Random rnd = new Random();
            originalArray = new int[size];
            for (int i = 0; i < size; i++)
            {
                originalArray[i] = rnd.Next(min, max + 1);
            }
            sortedArray = null; // сбрасываем отсортированный массив

            DisplayArray(originalArray, lstOriginal);
            lstSorted.Items.Clear();

            // Очищаем результаты предыдущих расчётов
            lblNegCount.Text = "-";
            lblPosCount.Text = "-";
            lblMinValue.Text = "-";
            lblMaxValue.Text = "-";
            lblSumPositive.Text = "-";
        }

        // Вывод массива в ListBox
        private void DisplayArray(int[] array, ListBox listBox)
        {
            listBox.Items.Clear();
            if (array == null || array.Length == 0)
                return;

            const int maxDisplay = 50; // Не показываем > 50 элементов
            if (array.Length <= maxDisplay)
            {
                foreach (int item in array)
                    listBox.Items.Add(item);
            }
            else
            {
                for (int i = 0; i < maxDisplay / 2; i++)
                    listBox.Items.Add(array[i]);
                listBox.Items.Add("...");
                for (int i = array.Length - maxDisplay / 2; i < array.Length; i++)
                    listBox.Items.Add(array[i]);
                listBox.Items.Add($"(всего элементов: {array.Length})");
            }
        }

        // Сортировка пузырьком
        private void btnSort_Click(object sender, EventArgs e)
        {
            if (originalArray == null || originalArray.Length == 0)
            {
                MessageBox.Show("Сначала сгенерируйте массив.", "Информация",
                                MessageBoxButtons.OK, MessageBoxIcon.Information);
                return;
            }

            sortedArray = BubbleSort(originalArray);
            DisplayArray(sortedArray, lstSorted);

            // Автоматически выполняем расчёты по исходному массиву
            PerformCalculations(originalArray);
        }

        private int[] BubbleSort(int[] arr)
        {
            int[] result = (int[])arr.Clone();
            int n = result.Length;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = 0; j < n - i - 1; j++)
                {
                    if (result[j] > result[j + 1])
                    {
                        // Обмен
                        int temp = result[j];
                        result[j] = result[j + 1];
                        result[j + 1] = temp;
                    }
                }
            }
            return result;
        }

        // Расчёт характеристик по массиву
        private void PerformCalculations(int[] array)
        {
            if (array == null || array.Length == 0) return;

            int negCount = array.Count(x => x < 0);
            int posCount = array.Count(x => x > 0);
            int minVal = array.Min();
            int maxVal = array.Max();
            int sumPos = array.Where(x => x > 0).Sum();

            lblNegCount.Text = negCount.ToString();
            lblPosCount.Text = posCount.ToString();
            lblMinValue.Text = minVal.ToString();
            lblMaxValue.Text = maxVal.ToString();
            lblSumPositive.Text = sumPos.ToString();
        }

        // Кнопка «Вычислить»
        private void btnCalculate_Click(object sender, EventArgs e)
        {
            if (originalArray == null || originalArray.Length == 0)
            {
                MessageBox.Show("Сначала сгенерируйте массив.", "Информация",
                                MessageBoxButtons.OK, MessageBoxIcon.Information);
                return;
            }
            PerformCalculations(originalArray);
        }
    }
}
